/**
 * Denne funktion henter data fra DB-tabellen categories
 */
exports.getCategories = async function() {
    try {
        const sql = `SELECT id, name, description FROM categories`;
        return await db.query(sql);
    } catch (error) {
        console.log(error);
        return null;
    }
};
